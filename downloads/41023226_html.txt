    try:
        url = doc.query["src"]
    except:
        url = "https://gist.githubusercontent.com/41023226/3eecdaab63826f23fc59577b61f05a1e/raw/9025db3d3f46fbad66ca7964626a6559cd04ac79/1%2520add%2520to%252010"

    # 按下 run 按鈕
    Ace1.run()
