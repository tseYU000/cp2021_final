    try:
        url = doc.query["src"]

        url = "https://gist.githubusercontent.com/mdecourse/0229a8a017091476a79700b8a190f185/raw/c3a6deaf717f8f2739a4b1392a5ab10936e9693a/from_1_add_to_10_1.py"

    Ace1.editor.gotoLine(0)

    # 按下 run 按鈕
